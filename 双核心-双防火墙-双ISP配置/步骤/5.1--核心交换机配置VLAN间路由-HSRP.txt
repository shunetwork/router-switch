CORE-SW1 
***************************************************
vlan10 20的主交换机是 sw1 备份 sw2
vlan50 90的主交换机是 sw2 备份 sw1


interface Vlan10
 ip address 192.168.10.3 255.255.255.0
 ip helper-address 10.11.11.38
 standby 10 ip 192.168.10.1
 standby 10 preempt 
 standby 10 priority 120
 standby 10 track g1/0/1
 standby 10 track g1/0/2
 exit

interface Vlan20
 ip address 172.16.0.3 255.255.0.0
 ip helper-address 10.11.11.38
 standby 20 ip 172.16.0.1
 standby 20 preempt 
 standby 20 priority 120
 standby 20 track g1/0/1
 standby 20 track g1/0/2
 exit
